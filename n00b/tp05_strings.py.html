def myindex(t, reverse=False):

    ival = raw_input("Enter a value: ")

    if reverse:
        l = range( len(t)-1, 0, -1 )
    else:
        l = range( len(t) )

    for i in l:
        if int(ival) == t[i]:
            print( i+1 )
            return
    print('The value was not found')

            print 'Mmmh weird'
            c =  'e'
        t += chr( ord(c) +  uppercase_distance )
